    Challenge 01¶

    Solução I¶

    Abaixo temos uma solução com as funções map, filter e reduce.

    from functools import reduce

funcionario_01 = {'Nome': "Marcos Paulo", 'Genero': "M", 'Salario': 1000.50, 'Setor': "TI"}
funcionario_02 = {'Nome': "Paulo Gustavo", 'Genero': "M", 'Salario': 3650.30, 'Setor': "RH"}
funcionario_03 = {'Nome': "Gustavo Alberto", 'Genero': "M", 'Salario': 5000.65, 'Setor': "TI"}
funcionario_04 = {'Nome': "Ana Paula", 'Genero': "F", 'Salario': 10000.50, 'Setor': "RH" }
funcionario_05 = {'Nome': "Paula", 'Genero': "F", 'Salario': 6000.50, 'Setor': "TI"}

bancoDeDados = [funcionario_01, funcionario_02, funcionario_03, funcionario_04, funcionario_05]

'Geral'
somaSalarioG = reduce(lambda soma, func: soma + func['Salario'],bancoDeDados,0)
mediaSalarioG = somaSalarioG / len(bancoDeDados)

print({'Geral':{'Soma':somaSalarioG,'Media':mediaSalarioG}})

'Homens'
funcHomens = list(filter(lambda func: True if func['Genero'] == 'M'else False,bancoDeDados))

somaSalarioH = reduce(lambda soma, salario: soma + salario,map(lambda func: func['Salario'],funcHomens),0)

qtdHomens = len(list(funcHomens))

mediaSalarioH = somaSalarioH / qtdHomens

'Mulheres'
somaSalarioM = reduce(lambda soma, func: soma + func['Salario'] if func['Genero'] == 'F' else soma,bancoDeDados,0)

qtdMulher = len(list(filter(lambda func: True if func['Genero'] == 'F' else False, bancoDeDados)))

mediaSalarioM = somaSalarioM / qtdMulher

print({
        'F':{'Soma':somaSalarioM,'Media':mediaSalarioM},
        'M':{'Soma':somaSalarioH,'Media':mediaSalarioH}
})

'Setores'

setores = set(map(lambda func: func['Setor'],bancoDeDados))


somaSalarioSetor = lambda setor: reduce(lambda soma, func: soma + func['Salario'],filter(lambda func: func['Setor'] == setor,bancoDeDados),0)
mediaSalarioSetor = lambda setor: somaSalarioSetor(setor) / len(list(filter(lambda func: func['Setor'] == setor,bancoDeDados)))

print({
    'RH':{'Soma':somaSalarioSetor('RH'),'Media':mediaSalarioSetor('RH')},
    'TI':{'Soma':somaSalarioSetor('TI'),'Media':mediaSalarioSetor('TI')}
})
    Comentários¶

    Solução ruim, se sua solução se parece com essa seria interessante reescrevela apurando a lógica e utilizando funções para repetir menos código, houveram abusos no uso de expressões lambda por conta de um uso também abusivo de map, filter e reduce, por mais que sua solução fucione tenha em mente que esse é um exercicio simples e deve ser resolvido de maneira concisa e objetiva.

    Solução II¶

    Abaixo temos uma solução utilizando groupby.

    from itertools import groupby
from operator import itemgetter

funcionario_01 = {'Nome': "Marcos Paulo", 'Genero': "M", 'Salario': 1000.50, 'Setor': "TI"}
funcionario_02 = {'Nome': "Paulo Gustavo", 'Genero': "M", 'Salario': 3650.30, 'Setor': "RH"}
funcionario_03 = {'Nome': "Gustavo Alberto", 'Genero': "M", 'Salario': 5000.65, 'Setor': "TI"}
funcionario_04 = {'Nome': "Ana Paula", 'Genero': "F", 'Salario': 10000.50, 'Setor': "RH" }
funcionario_05 = {'Nome': "Paula", 'Genero': "F", 'Salario': 6000.50, 'Setor': "TI"}

bancoDeDados = [funcionario_01, funcionario_02, funcionario_03, funcionario_04, funcionario_05]

'Geral'
somaSalarioG = reduce(lambda soma, func: soma + func['Salario'],bancoDeDados,0)
mediaSalarioG = somaSalarioG / len(bancoDeDados)

print({'Geral':{'Soma':somaSalarioG,'Media':mediaSalarioG}})

'Genero'
bancoDeDados.sort(key = lambda func: func['Genero'])
baseAgrupada = groupby(bancoDeDados,itemgetter('Genero'))
GeneroSalario = {f'{genero}': { 'Soma': sum(func['Salario'] for func in funcs)} for genero, funcs in baseAgrupada}
for key,value in GeneroSalario.items():
    value.update({'Media':value['Soma']/len(list(filter(lambda func: func['Genero'] == key,bancoDeDados)))})
print(GeneroSalario)

'Setor'
bancoDeDados.sort(key = lambda func: func['Setor'])
baseAgrupada = groupby(bancoDeDados,itemgetter('Setor'))
SetorSalario = {f'{setor}': {'Soma':sum(func['Salario'] for func in funcs)} for setor, funcs in baseAgrupada}
for key,value in SetorSalario.items():
    value.update({'Media':value['Soma']/len(list(filter(lambda func: func['Setor'] == key,bancoDeDados)))})
print(SetorSalario)
    Comentários¶

    Solução quase perfeita, acredito que o único defeito é que houve a violação de um principio chamado Don`t Repeat Yourself - DRY, percebam que quando fomos calcular as informações por setor e por genero escrevemos exatamente o mesmo algoritmo, ou seja, podemos parametriza-lo e tranformar aquele trecho de código em uma função.

    Solução III¶

    Abaixo solução com groupby e abstração em funções.

    from operator import itemgetter
from itertools import groupby
from itertools import _grouper
from functools import reduce
import locale

locale.setlocale(locale.LC_ALL, 'pt_BR.UTF-8')

funcionario_01 = {'Nome': "Marcos Paulo", 'Genero': "M", 'Salario': 1000.50, 'Setor': "TI"}
funcionario_02 = {'Nome': "Paulo Gustavo", 'Genero': "M", 'Salario': 3650.30, 'Setor': "RH"}
funcionario_03 = {'Nome': "Gustavo Alberto", 'Genero': "M", 'Salario': 5000.65, 'Setor': "TI"}
funcionario_04 = {'Nome': "Ana Paula", 'Genero': "F", 'Salario': 10000.50, 'Setor': "RH" }
funcionario_05 = {'Nome': "Paula", 'Genero': "F", 'Salario': 6000.50, 'Setor': "TI"}

bancoDeDados = [funcionario_01, funcionario_02, funcionario_03, funcionario_04, funcionario_05]

sumSalary = lambda data: reduce(lambda soma,item: soma + item['Salario'],data,0)
averageSalary = lambda soma,data: soma/len(data)

def info(items: _grouper) -> dict:
    data = list(items)
    soma = sumSalary(data)
    return {"Soma":locale.currency(soma,symbol=True),"Média":locale.currency(soma/len(data),symbol=True)}

def SalaryGroupBy(group: str, dataBase: list) -> dict:
    dataBase.sort(key = lambda item: item[group])
    groupDataBase = groupby(dataBase,itemgetter(group))
    return {f'{key}' : info(items) for key,items in groupDataBase}

'Geral'
print({'Geral':info(bancoDeDados)})

'Genero'
print(SalaryGroupBy('Genero',bancoDeDados))

'Setor'
print(SalaryGroupBy('Setor',bancoDeDados))
    Comentários¶

    Ao meu ver a solução perfeita para o desafio, se você chegou em algo melhor por favor compartilhe com algum instrutor, podemos posta-la aqui com seu nome se quiser U.u, percebam que essa ultima solução tem direito até a formatação dos valores em R$ kkkkk, o ponto mais importante é que evitamos cair no DRY quando abstraimos o código criando as nossas funções, não abusamos de expressões lambda e da triplice map,filter e reduce, outro ponto é que semânticamente nosso código é bem mais interessante que as outras duas soluções.
